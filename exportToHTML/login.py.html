import mysql.connector
import random

# 🔹 Conexión
conexion = mysql.connector.connect(
    host="localhost",
    user="root",
    password="",
    database="recomendacion",
    charset="utf8"
)

cursor = conexion.cursor()

print("=== SISTEMA DE LOGIN (escribe 'exit' para salir) ===")

while True:
    username = input("\nIngresa tu usuario: ")

    if username.lower() == "exit":
        print("Saliendo del sistema...")
        break

    password = input("Ingresa tu contraseña: ")

    #  Login
    sql = "SELECT * FROM usuario WHERE username = %s AND contraseña = %s"
    cursor.execute(sql, (username, password))
    resultado = cursor.fetchone()

    if resultado:
        print("Login correcto")

        tipo = resultado[3]


        if tipo == 1:
            print("\nPANEL ADMIN - TODOS LOS LIBROS\n")

            cursor.execute("SELECT id_libro, nombre, autor FROM libros")
            libros = cursor.fetchall()

            for libro in libros:
                print(f"ID: {libro[0]} | {libro[1]} - {libro[2]}")


        elif tipo == 2:
            print("\n PANEL USER - LIBROS ALEATORIOS\n")

            cursor.execute("SELECT id_libro, nombre, autor FROM libros")
            libros = cursor.fetchall()

            if len(libros) == 0:
                print(" No hay libros en la base de datos")
                continue

            cantidad = min(5, len(libros))
            seleccion = random.sample(libros, cantidad)

            print("Libros recomendados:\n")

            for i, libro in enumerate(seleccion, start=1):
                print(f"{i}. {libro[1]} - {libro[2]} (ID:{libro[0]})")

    else:
        print("Usuario o contraseña incorrectos")

# 🔹 Cerrar conexión
cursor.close()
conexion.close()
